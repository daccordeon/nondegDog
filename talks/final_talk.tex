\begin{frame}[noframenumbering]{Parity-time (PT) symmetry}
\centering
{\Large\begin{equation}
\hat{H}_I=i\hbar\omega_s(\hat{a}\hat{b}^\dag-\hat{a}^\dag\hat{b})+i\hbar\chi(\hat{b}^\dag\hat{c}^\dag-\hat{b}\hat{c})
\end{equation}}
\begin{enumerate}
\item parity: $\hat a\leftrightarrow \hat{c}$
\item time: $\hat a\leftrightarrow \hat a^\dag,\hat {c}\leftrightarrow \hat {c}^\dag$
\item parity-time: $\hat a \mapsto \hat c^\dag$ (and $\hat b \mapsto \hat b$)
\item $\hat H_I$ parity-time symmetric at $\omega_s=\chi$
\end{enumerate}
\end{frame}
